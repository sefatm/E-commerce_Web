Rural Mart Angular Frontend Structure

src/app/
├── core/
│   ├── guards/              # AuthGuard, AdminGuard
│   ├── interceptors/        # JWT interceptor
│   ├── models/              # TypeScript interfaces/models
│   └── services/            # API and state services
├── shared/
│   └── components/          # Reusable UI: site header, category nav, product card, cart modal
├── layout/                  # Admin/Seller layout, sidebar, navbar, footer
├── features/
│   ├── store/               # Public/customer pages: home, login, register, checkout, profile
│   └── admin/               # Admin and seller management screens
├── app-routing.module.ts
├── app.module.ts
└── app.component.*
